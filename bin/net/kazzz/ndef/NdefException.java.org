package net.kazzz.ndef;

/**
 * Ndef 例外クラスを提供します
 * 
 * @author Kazzz
 * @date 2011/01/21
 * @since Android API Level 9
 *
 */

public class NdefException extends Exception {
    private static final long serialVersionUID = 1L;
    /**
     * コンストラクタ
     * @param message メッセージをセット
     */
    public NdefException(String message) {
        super(message);
    }
    /**
     * コンストラクタ
     * @param cause 元の例外をセット
     */
    public NdefException(Exception cause) {
        super(cause);
    }
}
